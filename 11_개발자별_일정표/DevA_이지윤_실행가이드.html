import java.time.LocalDateTime;

public class AuthDtos {

    // ── Request DTOs ──
    public record LoginRequest(
        @NotBlank @Email String email,
        @NotBlank @Size(min = 8, max = 100) String password
    ) {}

    public record RegisterRequest(
        @NotBlank @Email @Size(max = 255) String email,
        @NotBlank @Pattern(
            regexp = "^(?=.*[A-Za-z])(?=.*\\d)(?=.*[!@#$%^&*]).{8,}$",
            message = "영문+숫자+특수문자 포함 8자 이상"
        )
        String password,
        @NotBlank @Size(min = 2, max = 50) String nickname
    ) {}

    public record RefreshRequest(
        @NotBlank String refreshToken
    ) {}

    public record LogoutRequest(
        @NotBlank String refreshToken
    ) {}

    public record OAuthGoogleRequest(
        @NotBlank String idToken
    ) {}

    // ── Response DTOs ──
    public record TokenResponse(
        String accessToken,
        String refreshToken,
        long   expiresIn   // 생명—초 단위 (Access Token: 1800)
    ) {}

    public record UserResponse(

        String         email,
        String         nickname,
        String         role,         // ROLE_USER

    ) {}

    public record RefreshTokenDto(
        String         jti,
        String         token,
        LocalDateTime  expiresAt
    ) {}
}

5.3글로벌 익셉션 핸들러 + ProblemDetail (RFC 7807)
⚠️ 04번 §2.4 내용 그대로 구현. 모든 서비스 동일 패턴.
// auth-service/.../common/exception/BusinessException.java
@Getter
public class BusinessException extends RuntimeException {
    private final ErrorCode error;

    public BusinessException(ErrorCode error) {
        super(error.getMessage());
        this.error = error;
    }
}

// auth-service/.../common/exception/GlobalExceptionHandler.java
@RestControllerAdvice
@Slf4j
public class GlobalExceptionHandler {

    // 도메인 에러 — 04번 §2.4 ProblemDetail
    @ExceptionHandler(BusinessException.class)
    public ProblemDetail handleBusinessException(BusinessException e, HttpServletRequest req) {
        ErrorCode err = e.getError();
        ProblemDetail pd = ProblemDetail.forStatus(err.getHttpStatus());
        pd.setType(URI.create("https://api.qtai.app/errors/" + err.getCode().toLowerCase().replace('_', '-')));
        pd.setTitle(err.getMessage());
        pd.setDetail(err.getMessage());
        pd.setInstance(URI.create(req.getRequestURI()));
        pd.setProperty("code",      err.getCode());
        pd.setProperty("traceId",   MDC.get("traceId"));
        pd.setProperty("timestamp", Instant.now().toString());
        return pd;
    }

    // Bean Validation 에러 — errors[] 배열 포함
    @ExceptionHandler(MethodArgumentNotValidException.class)
    public ProblemDetail handleValidation(MethodArgumentNotValidException e, HttpServletRequest req) {
        ProblemDetail pd = ProblemDetail.forStatus(HttpStatus.BAD_REQUEST);
        pd.setTitle("Validation failed");
        pd.setProperty("code",    "VALIDATION_FAILED");
        pd.setProperty("traceId", MDC.get("traceId"));
        pd.setProperty("timestamp", Instant.now().toString());
        pd.setProperty("errors",
            e.getBindingResult().getFieldErrors().stream()
                .map(fe -> Map.of(
                    "field",   fe.getField(),
                    "code",    fe.getCode(),
                    "message", fe.getDefaultMessage()
                )).toList()
        );
        return pd;
    }

    // Circuit Breaker OPEN — 04번 §2.5 503 명시 매핑
    @ExceptionHandler(CallNotPermittedException.class)
    public ProblemDetail circuitBreakerOpen(CallNotPermittedException e) {
        ProblemDetail pd = ProblemDetail.forStatus(HttpStatus.SERVICE_UNAVAILABLE);
        pd.setProperty("code",    "UPSTREAM_UNAVAILABLE");
        pd.setProperty("traceId", MDC.get("traceId"));
        return pd;
    }
}

5.4Kafka 페이로드 스키마 — user.deactivated
⚠️ Auth Service가 발행하는 유일한 이벤트. events/schema/user.deactivated-value.json과 완전 동일. Apicurio Schema Registry에 등록 필수.
// auth-service/.../infrastructure/kafka/event/UserDeactivatedEvent.java
public record UserDeactivatedEvent(
    String         eventId,
    String         eventType,        // "user.deactivated"
    int            eventVersion,     // 1
    String         schemaSubject,    // "user.deactivated-value"
    String         occurredAt,       // ISO 8601
    String         traceId,
    String         producerService,  // "auth-service"
    String         idempotencyKey,   // "user.deactivated:{userId}"
    Data           data
) {
    public record Data(
        long   userId,
        String deactivatedAt   // ISO 8601
    ) {}

    public static UserDeactivatedEvent of(long userId, LocalDateTime deactivatedAt) {
        return new UserDeactivatedEvent(
            UUID.randomUUID().toString(),
            "user.deactivated",
            1,
            "user.deactivated-value",
            Instant.now().toString(),
            MDC.get("traceId"),
            "auth-service",
            "user.deactivated:" + userId,
            new Data(userId, deactivatedAt.toString())
        );
    }
}

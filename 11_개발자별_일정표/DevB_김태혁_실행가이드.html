    public record CommentaryResponse(
        String                  bookCode,
        int                     chapter,
        int                     verse,
        List    items   // 0~N개. 빈 배열 OK (404 아님)
    ) {
        public record CommentaryItem(
            Long   id,
            String source,    // MATTHEW_HENRY / DUMMY_KR
            String language,  // "en" / "ko"
            String title,
            String content
        ) {}
    }

    // ── GET /bible/books 응답 ──
    public record BookListResponse(
        List items  // 66권 고정, 페이지네이션 없음

            String code,      // GEN, EXO, ..., REV
            String nameKr,    // 창세기
            String nameEn,    // Genesis
            int    chapters
        ) {}

5.3글로벌 익셉션 핸들러 + ProblemDetail (RFC 7807) 
// bible-service/.../common/exception/GlobalExceptionHandler.java
@RestControllerAdvice
@Slf4j
public class GlobalExceptionHandler {

    @ExceptionHandler(BusinessException.class)
    public ProblemDetail handleBusinessException(BusinessException e, HttpServletRequest req) {
        ErrorCode err = e.getError();
        ProblemDetail pd = ProblemDetail.forStatus(err.getHttpStatus());
        pd.setType(URI.create("https://api.qtai.app/errors/" + err.getCode().toLowerCase().replace('_', '-')));
        pd.setTitle(err.getMessage());
        pd.setInstance(URI.create(req.getRequestURI()));
        pd.setProperty("code",      err.getCode());
        pd.setProperty("traceId",   MDC.get("traceId"));
        pd.setProperty("timestamp", Instant.now().toString());
        return pd;
    }

    @ExceptionHandler(MethodArgumentNotValidException.class)
    public ProblemDetail handleValidation(MethodArgumentNotValidException e, HttpServletRequest req) {
        ProblemDetail pd = ProblemDetail.forStatus(HttpStatus.BAD_REQUEST);
        pd.setProperty("code",     "VALIDATION_FAILED");
        pd.setProperty("traceId",  MDC.get("traceId"));
        pd.setProperty("timestamp", Instant.now().toString());
        pd.setProperty("errors",
            e.getBindingResult().getFieldErrors().stream()
                .map(fe -> Map.of(
                    "field",   fe.getField(),
                    "code",    fe.getCode(),
                    "message", fe.getDefaultMessage()
                )).toList()
        );
        return pd;
    }
}

5.6Bible DB 스키마 (V1 SQL)
Bible DB는 4개 테이블: BOOKS·KR_BIBLE·EN_BIBLE·COMMENTARIES.
-- V1__create_bible_tables.sql
CREATE TABLE BOOKS (
    id          BIGINT      AUTO_INCREMENT PRIMARY KEY,
    code        VARCHAR(3)  NOT NULL UNIQUE,    -- GEN, EXO, ..., REV
    name_kr     VARCHAR(20) NOT NULL,           -- 창세기
    name_en     VARCHAR(30) NOT NULL,           -- Genesis
    chapters    INT         NOT NULL,           -- 50
    sort_order  INT         NOT NULL,           -- 1~66
    INDEX idx_books_code (code)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE KR_BIBLE (
    id          BIGINT       AUTO_INCREMENT PRIMARY KEY,
    book_id     BIGINT       NOT NULL,
    chapter     INT          NOT NULL,
    verse       INT          NOT NULL,
    content     TEXT         NOT NULL,
    version     VARCHAR(20)  NOT NULL DEFAULT 'REVISED',  -- 개역한글
    UNIQUE KEY uk_kr_coord (book_id, chapter, verse),
    INDEX      idx_kr_book_chap (book_id, chapter),
    FOREIGN KEY (book_id) REFERENCES BOOKS(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE EN_BIBLE (
    id          BIGINT       AUTO_INCREMENT PRIMARY KEY,
    book_id     BIGINT       NOT NULL,
    chapter     INT          NOT NULL,
    verse       INT          NOT NULL,
    content     TEXT         NOT NULL,
    version     VARCHAR(20)  NOT NULL DEFAULT 'KJV',
    UNIQUE KEY uk_en_coord (book_id, chapter, verse),
    INDEX      idx_en_book_chap (book_id, chapter),
    FOREIGN KEY (book_id) REFERENCES BOOKS(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE COMMENTARIES (
    id          BIGINT       AUTO_INCREMENT PRIMARY KEY,
    book_id     BIGINT       NOT NULL,
    chapter     INT          NOT NULL,
    verse       INT          NOT NULL,
    source      VARCHAR(30)  NOT NULL,         -- MATTHEW_HENRY / DUMMY_KR
    language    VARCHAR(2)   NOT NULL,         -- ko / en
    title       VARCHAR(255) NOT NULL,
    content     TEXT         NOT NULL,
    INDEX idx_comm_coord (book_id, chapter, verse),
    FOREIGN KEY (book_id) REFERENCES BOOKS(id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

• 다른 팀원 폴더 변경이 PR에 포함되면 자동 reject

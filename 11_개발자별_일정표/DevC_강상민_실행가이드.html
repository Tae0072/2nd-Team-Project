📜 이 문서만 보고 작업하는 규칙

  1. 매일 본인 W1~W5 카드만 먼저 펼쳐서 오늘 할 일을 파악합니다.
  2. 작업 브랜치는 feature/ai-{task-slug} 형식 (예: feature/ai-sse-stream).
  3. 30분 이상 막히면 혼자 붙잡지 않고 Lead(강태오)에게 ping합니다.
  4. 작업 시작 전 workspaces/{본인}/workflows/에 워크플로우 작성, 종료 후 reports/에 리포트 작성 (S6 §6.2 / §6.3).
  5. PR은 하루 종료 전 draft로 등록, 완료 기준은 각 day 카드의 📋 DoD를 따릅니다.
  6. 다른 팀원의 workspaces/ 폴더는 읽지·수정·삭제하지 않습니다 (AGENTS.md §13).
  7. 코드/스키마 변경이 필요하면 S7 원문 문서 매트릭스를 먼저 확인 후 해당 원본 문서를 봅니다.

5.1Pydantic 모델 — Request/Response 명세
⚠️ Python은 snake_case가 관습이지만 API 응답은 camelCase. alias_generator=to_camel로 둘 다 지원.
# ai-service/app/domain/schemas.py
from pydantic import BaseModel, ConfigDict, Field
from pydantic.alias_generators import to_camel
from datetime import datetime
from typing import Optional, List, Literal


class CamelModel(BaseModel):
    model_config = ConfigDict(
        alias_generator=to_camel,
        populate_by_name=True,
        from_attributes=True
    )

# ── Path/Body 입력 ──
class PassageRef(CamelModel):
    book_code: str = Field(..., min_length=3, max_length=3)  # GEN, EXO, ..., REV
    chapter:   int = Field(..., ge=1)
    verse:     int = Field(..., ge=1)


class StartSessionRequest(CamelModel):
    passage:      PassageRef
    initial_step: Literal["A", "B", "C", "D"] = "A"  # default "A" (04번 §6.2)


class TurnRequest(CamelModel):
    user_message: str = Field(..., min_length=1, max_length=4000)


# ── 응답 ──
class PassageDetail(CamelModel):
    book_id:   int
    book_code: str
    chapter:   int
    verse:     int


class SessionResponse(CamelModel):
    id:           int
    user_id:      int
    passage:      PassageDetail
    current_step: Literal["A", "B", "C", "D"]
    status:       Literal["IN_PROGRESS", "COMPLETED", "ABANDONED", "COMPLETED_NO_JOURNAL"]
    summary:      Optional[str] = None
    created_at:   datetime
    updated_at:   Optional[datetime] = None


class TurnDetail(CamelModel):
    id:                 int
    role:               Literal["SYSTEM", "USER", "ASSISTANT"]
    step:               Optional[Literal["A", "B", "C", "D"]] = None
    content:            str          # SYSTEM은 언제나 ""
    content_redacted:   bool         # SYSTEM은 true → 클라이언트에서 숨김
    prompt_template_id: Optional[int] = None
    rag_sources:        Optional[List[dict]] = None
    created_at:         datetime


class SessionDetailResponse(SessionResponse):
    turns: List[TurnDetail]


# ── 페이지네이션 ──
class PageInfo(CamelModel):
    number:         int
    size:           int
    total_elements: int
    total_pages:    int


class SessionListResponse(CamelModel):
    items: List[SessionResponse]
    page:  PageInfo

#   turn_started, token, rag_sources, step_changed, turn_completed, end, error
# ai-service/app/api/v1/turns.py
from fastapi import APIRouter, Depends, HTTPException, Header
from sse_starlette.sse import EventSourceResponse
import asyncio, json

router = APIRouter()


@router.post("/ai/sessions/{session_id}/turns")
async def add_turn(
    session_id:  int,
    body:        TurnRequest,
    x_user_id:   int = Header(..., alias="X-User-Id"),
    chat_uc:     ChatUseCase = Depends(),
):
    # ── 스트림 시작 전 HTTP 에러 (404/403/422/502/504) ──
    session = chat_uc.validate_and_load(session_id, x_user_id)

    async def event_generator():
        try:
            yield {"event": "turn_started", "data": json.dumps({
                "turnId": None, "step": session.current_step,
                "timestamp": datetime.utcnow().isoformat() + "Z"
            })}

            # Anthropic 스트림
            async for token in chat_uc.stream(session, body.user_message):
                yield {"event": "token", "data": json.dumps({"text": token})}

            # RAG 출처 — 매 ASSISTANT 턴마다 필수 (03번 §3.3)
            yield {"event": "rag_sources", "data": json.dumps({"sources": chat_uc.rag_sources})}

            # 턴 완료 — DB 적재 후
            saved = chat_uc.save_assistant_turn(...)
            yield {"event": "turn_completed", "data": json.dumps({
                "turnId":           saved.id,
                "step":             saved.step,
                "promptTemplateId": saved.prompt_template_id,
                "totalTokens":      saved.total_tokens,
                "elapsedMs":        saved.elapsed_ms
            })}
        except Exception as e:
            # error 도중 발생 — 04번 §6.3 error→end 정책
            yield {"event": "error", "data": json.dumps({
                "code": classify_error(e),
                "detail": str(e),
                "traceId": get_trace_id()
            })}
        finally:
            # 반드시 end 발행 (클라이언트 정상 종료 신호)
            yield {"event": "end", "data": "[DONE]"}

    return EventSourceResponse(event_generator())

5.4Kafka 페이로드 — ai.session.completed (발행)
⚠️ D 단계 도달 시 AFTER_COMMIT에 발행. confluent-kafka + Apicurio Schema Registry 사용. events/schema/ai.session.completed-value.json과 완전 동일.
# ai-service/app/kafka/event_publisher.py
from confluent_kafka import Producer
from datetime import datetime, timezone
import ulid, json


def build_ai_session_completed_event(
    session_id: int,
    user_id:    int,
    passage:    dict,    # {"bookId":1,"bookCode":"GEN","chapter":1,"verse":1}
    final_step: str,     # "D"
    summary:    str,
    trace_id:   str
) -> dict:
    """events/schema/ai.session.completed-value.json과 완전 동일."""
    return {
        "eventId":         f"evt_{ulid.new().str}",
        "eventType":       "ai.session.completed",
        "eventVersion":    1,
        "schemaSubject":   "ai.session.completed-value",
        "occurredAt":      datetime.now(timezone.utc).isoformat(),
        "traceId":         trace_id,
        "producerService": "ai-service",
        "idempotencyKey":  f"ai.session.completed:{session_id}",
        "data": {
            "sessionId": session_id,
            "userId":    user_id,
            "passage":   passage,
            "finalStep": final_step,
            "summary":   summary
        }
    }


producer = Producer({"bootstrap.servers": "kafka.qtai:9092", "acks": "all"})


def publish_session_completed(session, summary: str, trace_id: str):
    """AFTER_COMMIT 훅에서 호출 (SQLAlchemy event listener 또는 outbox)."""
    event = build_ai_session_completed_event(
        session_id = session.id,
        user_id    = session.user_id,
        passage    = {
            "bookId":   session.book_id,
            "bookCode": session.book_code,
            "chapter":  session.chapter,
            "verse":    session.verse
        },
        final_step = "D",
        summary    = summary,
        trace_id   = trace_id
    )
    producer.produce(
        topic = "ai.session.completed",
        key   = event["idempotencyKey"].encode(),
        value = json.dumps(event).encode()
    )
    producer.flush(timeout=2)

import json


# ── 1) user.deactivated ──
def handle_user_deactivated(event: dict):
    """해당 user의 IN_PROGRESS 세션을 ABANDONED로 마킹."""
    user_id = event["data"]["userId"]
    db.execute(
        "UPDATE AI_SESSIONS SET status='ABANDONED' "
        "WHERE user_id = :uid AND status = 'IN_PROGRESS'",
        {"uid": user_id}
    )
    db.commit()


# ── 2) journal.creation.failed (Saga 보상) ──
def handle_journal_creation_failed(event: dict):
    """AI_SESSIONS.status → 'COMPLETED_NO_JOURNAL' (멱등 체크 필수)."""
    session_id      = event["data"]["sessionId"]
    idempotency_key = event["idempotencyKey"]   # journal.creation.failed:{sessionId}

    # 멱등성 — 이미 처리된 이벤트 skip
    if processed_event_repo.exists(idempotency_key):
        return

    db.execute(
        "UPDATE AI_SESSIONS SET status='COMPLETED_NO_JOURNAL' "
        "WHERE id = :sid",
        {"sid": session_id}
    )
    processed_event_repo.save(idempotency_key)
    db.commit()


# Consumer 등록
consumer = Consumer({
    "bootstrap.servers": "kafka.qtai:9092",
    "group.id":          "ai-service-saga-compensation-consumer",
    "enable.auto.commit": False  # 명시적 commit
})
consumer.subscribe(["user.deactivated", "journal.creation.failed"])

while True:
    msg = consumer.poll(1.0)
    if msg is None: continue
    event = json.loads(msg.value())
    if event["eventType"] == "user.deactivated":
        handle_user_deactivated(event)
    elif event["eventType"] == "journal.creation.failed":
        handle_journal_creation_failed(event)
    consumer.commit(msg)

📝 workflows/{date}-{task}.md 작성 → 🔍 자기 검토 (체크리스트 ✓) → 🚀 작업 진행 → 📊 reports/{date}-{task}.md 작성
🚫 폴더 격리 규칙 (AGENTS.md §13)
• 본인 폴더(workspaces/DevC_강상민/)에서만 읽기·쓰기 허용
• 다른 팀원의 개인 폴더는 읽기·수정·삭제 모두 금지 (AI 에이전트 포함)
• workspaces/ 내 파일은 빌드·테스트·CI에 영향 없음 (import 금지)
• 다른 팀원 폴더 변경이 PR에 포함되면 자동 reject

📝 무엇을 (WHAT)

🛠 어떻게 (HOW)
@Component
public class DeepSeekStreamService {
    private final RestClient client;
    private final Semaphore concurrency = new Semaphore(8);  // 동시 8 stream

    public Flux<String> stream(String prompt, AtomicBoolean cancelled) {
        if (!concurrency.tryAcquire())
            throw new ProblemException("LLM_OVERLOADED");
        return client.post()
            .uri("/v1/chat/completions")
            .body(Map.of("model", "deepseek-chat", "stream", true,
                         "messages", List.of(Map.of("role","user","content",prompt))))
            .retrieve()
            .bodyToFlux(String.class)
            .takeUntil(s -> cancelled.get())
            .timeout(Duration.ofSeconds(30))
            .doFinally(sig -> concurrency.release());
    }
}

@UseCase
@RequiredArgsConstructor
public class CompleteSessionUseCase {
    private final AiSessionRepository repo;
    private final ApplicationEventPublisher events;

    @Transactional
    public void execute(Long sessionId, String summary) {
        var session = repo.findById(sessionId).orElseThrow();
        session.complete(summary);
        events.publishEvent(new AiSessionCompletedEvent(session.getId(),
            session.getQtDate(), summary, UUID.randomUUID()));
    }
}

@Component
public class AiSessionEventPublisher {
    @TransactionalEventListener(phase=TransactionPhase.AFTER_COMMIT)
    public void onCompleted(AiSessionCompletedEvent e) {
        kafkaTemplate.send("ai.session.completed",
            Map.of("data", e.toMap()));   // payload 키 금지, data만
    }
}

@UseCase
public class ProcessTurnUseCase {
    private final SystemPromptBuilder promptBuilder;     // DevB
    private final ChromaClient chroma;                   // DevB
    private final DeepSeekStreamService deepseek;        // 본인

    public void execute(Long sessionId, TurnRequest req, SseEventBuilder sse) {
        var session = sessionRepo.findById(sessionId).orElseThrow();
        var system = promptBuilder.build(req.stage(), session.getPassage());
        var sources = chroma.search(session.book, session.ch, session.v);
        sse.send("turn_started", session);
        deepseek.stream(system + req.userText())
            .doOnNext(token -> sse.send("token", token))
            .doOnComplete(() -> {
                sse.send("sources", sources);
                sse.send("turn_completed", finalSummary);
                sse.send("[DONE]");
            }).subscribe();
    }
}

# 해설 생성 파이프라인 (강상민 정의)
@Service
public class ExplanationGenerator {
    public Explanation generate(Verse verse) {
        String mdRef = referenceData.matthewHenry(verse);      // 비교 데이터
        String draft = deepseek.chat(buildSystemPrompt(verse, mdRef));
        EditorResult result = editor.verify(draft, verse, mdRef);
        if (result.isPassed()) {
            return Explanation.fromAi(draft, result.sourceIds());
        }
        return retry(verse) // 정책: 2회 재시도 → 이지윤 큐로 이관
            .orElseThrow(() -> new GenerationFailedException(verse));
    }
}

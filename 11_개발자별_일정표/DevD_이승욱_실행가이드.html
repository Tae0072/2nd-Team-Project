public interface JournalRepository extends JpaRepository<Journal, Long> {
    @Lock(LockModeType.PESSIMISTIC_WRITE)
    @Query("SELECT j FROM Journal j WHERE j.id = :id")
    Optional<Journal> findByIdForUpdate(Long id);
}

@UseCase
@Transactional
public class PatchJournalUseCase {
    public Journal execute(Long id, PatchRequest req) {
        var journal = journalRepo.findByIdForUpdate(id).orElseThrow();
        journal.patch(req);
        eventRepo.save(new JournalEvent(id, "PATCHED",
            req.changedFields(), req.idempotencyKey(), journal.nextSeq()));
        return journal;
    }
}

void concurrent_patch_keeps_sequence_monotonic() throws InterruptedException {
    int N = 50;
    var latch = new CountDownLatch(N);
    for (int i = 0; i < N; i++) {
        executor.submit(() -> {
            patchJournalUseCase.execute(journalId, randomPatch());
            latch.countDown();
        });
    }
    latch.await();
    var seqs = eventRepo.findAllByJournalIdOrderBySeq(journalId)
        .stream().map(JournalEvent::getSequence).toList();
    assertThat(seqs).isSortedAccordingTo(Long::compareTo);
}

    var key = "ai.session.completed:" + e.sessionId();
    try {
        inboxKeyRepo.saveAndFlush(new InboxKey(key, "ai.session.completed", now()));
    } catch (DuplicateKeyException dup) {
        log.info("duplicate event, skip: {}", key); return;
    }
    var journal = journalRepo.findByUserIdAndQtDate(e.userId(), e.qtDate())
        .orElseThrow(JournalNotFoundException::new);
    journal.attachAiSummary(e.aiSessionId(), e.summary());
    journalEventStore.append(JournalEvent.aiSummaryAttached(journal.id(), e));
}

# Journal PATCH 자동 저장 (debounce)
@PatchMapping("/{id}")
@Transactional
public ResponseEntity<Void> patch(@PathVariable Long id, @RequestBody PatchRequest req) {
    var j = journalRepo.findByIdForUpdate(id).orElseThrow();  // PESSIMISTIC_WRITE
    j.applyPatch(req);
    journalEventStore.append(JournalEvent.updated(id, req));
    return ResponseEntity.noContent().build();
}

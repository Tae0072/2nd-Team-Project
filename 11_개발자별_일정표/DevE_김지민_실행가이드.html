📜 이 문서만 보고 작업하는 규칙

  1. 매일 본인 W1~W5 카드만 먼저 펼쳐서 오늘 할 일을 파악합니다.
  2. 작업 브랜치는 feature/app-{task-slug} 형식 (예: feature/app-bible-screen).
  3. 30분 이상 막히면 혼자 붙잡지 않고 Lead(강태오)에게 ping합니다.
  4. 작업 시작 전 workspaces/{본인}/workflows/에 워크플로우 작성, 종료 후 reports/에 리포트 작성 (S6 §6.2 / §6.3).
  5. PR은 하루 종료 전 draft로 등록, 완료 기준은 각 day 카드의 📋 DoD를 따릅니다.
  6. 다른 팀원의 workspaces/ 폴더는 읽지·수정·삭제하지 않습니다 (AGENTS.md §13).
  7. 코드/스키마 변경이 필요하면 S7 원문 문서 매트릭스를 먼저 확인 후 해당 원본 문서를 봅니다.

시연에서 심사위원이 직접 손으로 만지는 유일한 인터페이스다. Sliver 기반 Sync Scroll과 AI SSE 스트리밍 실시간 타이핑 효과가 포트폴리오의 WOW 포인트가 된다. 백엔드 API가 완성되기 전에도 Mock 데이터로 화면을 미리 완성한다.

5.3Dio 클라이언트 + JWT Auto-Refresh Interceptor 
// flutter-app/lib/core/network/dio_client.dart
import 'package:dio/dio.dart';

class DioClient {
  static Dio create({required String baseUrl, required AuthRepository auth}) {
    final dio = Dio(BaseOptions(
      baseUrl: baseUrl,                   // https://api.qtai.app
      connectTimeout: Duration(seconds: 5),
      receiveTimeout: Duration(seconds: 30),
      headers: {'Content-Type': 'application/json'},
    ));

    // JWT 주입 + 401 자동 refresh
    dio.interceptors.add(InterceptorsWrapper(
      onRequest: (options, handler) async {
        final token = await auth.getAccessToken();
        if (token != null) {
          options.headers['Authorization'] = 'Bearer $token';
        }
        handler.next(options);
      },
      onError: (e, handler) async {
        // 401 → refresh 재시도 (1회만)
        if (e.response?.statusCode == 401 && !e.requestOptions.extra.containsKey('retried')) {
          try {
            await auth.refreshToken();
            final retryReq = e.requestOptions;
            retryReq.extra['retried'] = true;
            retryReq.headers['Authorization'] = 'Bearer ${await auth.getAccessToken()}';
            final response = await dio.fetch(retryReq);
            return handler.resolve(response);
          } catch (_) {
            await auth.logout();   // refresh 실패 → 로그아웃
          }
        }
        handler.next(e);
      },
    ));

    return dio;
  }
}
5.4SSE 클라이언트 — AI 턴 스트림
⚠️ 의존성: flutter_client_sse: ^2.0.0. event: end 도착 시 반드시 Stream 종료. error 이벤트도 end 후 상위 함수으로 종료.
// flutter-app/lib/features/ai/data/sse_client.dart
import 'package:flutter_client_sse/flutter_client_sse.dart';

class AiSseClient {
  Stream streamTurn({
    required int    sessionId,
    required String userMessage,
    required String accessToken,
  }) async* {
    final stream = SSEClient.subscribeToSSE(
      method:  SSERequestType.POST,
      url:     'https://api.qtai.app/ai/sessions/$sessionId/turns',
      header:  {
        'Authorization': 'Bearer $accessToken',
        'Content-Type':  'application/json',
        'Accept':        'text/event-stream',
      },
      body: {'userMessage': userMessage},
    );

    await for (final raw in stream) {
      if (raw.event == 'end') return;   // 종료

      switch (raw.event) {
        case 'turn_started':
          yield TurnStartedSseEvent.fromJson(jsonDecode(raw.data!));
          break;
        case 'token':
          yield TokenSseEvent(text: jsonDecode(raw.data!)['text']);
          break;
        case 'rag_sources':
          yield RagSourcesSseEvent.fromJson(jsonDecode(raw.data!));
          break;
        case 'turn_completed':
          yield TurnCompletedSseEvent.fromJson(jsonDecode(raw.data!));
          break;
        case 'error':
          final err = jsonDecode(raw.data!);
          yield ErrorSseEvent(
            code:    err['code'],
            detail:  err['detail'],
            traceId: err['traceId'],
          );
          // error 다음에 반드시 end 수신 → 자연 종료
          break;
      }
    }
  }
}

sealed class SseEvent {}
class TurnStartedSseEvent  extends SseEvent { final String step; final DateTime timestamp; ... }
class TokenSseEvent        extends SseEvent { final String text; }
class RagSourcesSseEvent   extends SseEvent { final List> sources; }
class TurnCompletedSseEvent extends SseEvent { final int turnId; final int totalTokens; ... }
class ErrorSseEvent        extends SseEvent { final String code; final String detail; final String traceId; }

// flutter-app/lib/features/notifications/stomp_client.dart
import 'package:stomp_dart_client/stomp_dart_client.dart';

class StompNotificationClient {
  StompClient? _client;
  final void Function(NotificationEvent) onMessage;

  StompNotificationClient({required this.onMessage});

  void connect({required int userId, required String accessToken}) {
    _client = StompClient(
      config: StompConfig.sockJS(
        url: 'https://api.qtai.app/ws/notifications',
        onConnect: (frame) => _subscribe(userId),
        stompConnectHeaders: {'Authorization': 'Bearer $accessToken'},
        webSocketConnectHeaders: {'Authorization': 'Bearer $accessToken'},
        onWebSocketError: (e) => print('WS error: $e'),
        // 명확한 reconnect 정책
        reconnectDelay: Duration(seconds: 5),
      ),
    );
    _client!.activate();
  }

  void _subscribe(int userId) {
    _client!.subscribe(
      destination: '/topic/user.$userId',
      callback: (frame) {
        if (frame.body == null) return;
        final json = jsonDecode(frame.body!);
        onMessage(NotificationEvent.fromJson(json));
      },
    );
  }

  void disconnect() => _client?.deactivate();
}

5.6Riverpod Auth Provider 패턴 
// flutter-app/lib/features/auth/auth_provider.dart
import 'package:flutter_riverpod/flutter_riverpod.dart';
import 'package:flutter_secure_storage/flutter_secure_storage.dart';

// 안전 저장소 — access/refresh 토큰
final secureStorageProvider = Provider((_) => const FlutterSecureStorage());

// 인증 상태
class AuthState {
  final UserResponse? user;
  final bool isAuthenticated;
  final bool isLoading;
  final String? errorCode;

  const AuthState({this.user, this.isAuthenticated = false, this.isLoading = false, this.errorCode});
}

class AuthNotifier extends StateNotifier {
  final AuthRepository repo;
  final FlutterSecureStorage storage;

  AuthNotifier(this.repo, this.storage) : super(const AuthState());

  Future login(String email, String password) async {
    state = AuthState(isLoading: true);
    try {
      final tokens = await repo.login(email, password);
      await storage.write(key: 'access',  value: tokens.accessToken);
      await storage.write(key: 'refresh', value: tokens.refreshToken);
      final user = await repo.getMe();
      state = AuthState(user: user, isAuthenticated: true);
    } on DioException catch (e) {
      final problem = ProblemDetail.fromJson(e.response?.data);
      state = AuthState(errorCode: problem.code);   // INVALID_CREDENTIALS 등
    }
  }

  Future getAccessToken() => storage.read(key: 'access');

  Future refreshToken() async {
    final refresh = await storage.read(key: 'refresh');
    if (refresh == null) throw Exception('No refresh token');
    final tokens = await repo.refresh(refresh);
    await storage.write(key: 'access',  value: tokens.accessToken);
    await storage.write(key: 'refresh', value: tokens.refreshToken);
  }

  Future logout() async {
    final refresh = await storage.read(key: 'refresh');
    if (refresh != null) await repo.logout(refresh);
    await storage.deleteAll();
    state = const AuthState();
  }
}

final authProvider = StateNotifierProvider((ref) {
  return AuthNotifier(ref.read(authRepositoryProvider), ref.read(secureStorageProvider));
});

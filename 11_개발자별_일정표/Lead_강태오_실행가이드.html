@Configuration
public class RouteConfig {

    @Bean
    public RouteLocator routes(RouteLocatorBuilder builder) {
        return builder.routes()
            .route("auth",     r -> r.path("/auth/**").uri("http://auth-service.qtai.svc.cluster.local:8080"))

5.4Kafka 페이로드 — user.activity.tracked (BFF 발행)
⚠️ BFF가 발행하는 유일한 이벤트. /api/v1/passages/... 조회 시 Best-effort 발행 (응답 중단 X). events/schema/user.activity.tracked-value.json과 완전 동일.
// bff-aggregator/.../infrastructure/kafka/event/UserActivityTrackedEvent.java
public record UserActivityTrackedEvent(
    String         eventId,
    String         eventType,        // "user.activity.tracked"
    int            eventVersion,     // 1
    String         schemaSubject,    // "user.activity.tracked-value"
    String         occurredAt,
    String         traceId,
    String         producerService,  // "bff-aggregator"
    String         idempotencyKey,   // "read.passage:{userId}:{book}:{ch}:{v}:{epochMinute}"
    Data           data
) {
    public record Data(
        long      userId,
        String    activityType,    // "READ_PASSAGE"
        Passage   passage,
        String    occurredAt
    ) {
        public record Passage(String bookCode, int chapter, int verse) {}
    }

    public static UserActivityTrackedEvent ofReadPassage(
        long userId, String bookCode, int chapter, int verse
    ) {
        long epochMinute = Instant.now().getEpochSecond() / 60;
        return new UserActivityTrackedEvent(
            UUID.randomUUID().toString(),
            "user.activity.tracked",
            1,
            "user.activity.tracked-value",
            Instant.now().toString(),
            MDC.get("traceId"),
            "bff-aggregator",
            String.format("read.passage:%d:%s:%d:%d:%d", userId, bookCode, chapter, verse, epochMinute),
            new Data(userId, "READ_PASSAGE", new Data.Passage(bookCode, chapter, verse), Instant.now().toString())
        );
    }
}

// 발행 예 (PassageAggregateUseCase 내부, Best-effort)
try {
    UserActivityTrackedEvent event = UserActivityTrackedEvent.ofReadPassage(userId, bookCode, chapter, verse);
    kafkaTemplate.send("user.activity.tracked", event.idempotencyKey(), event);
} catch (Exception e) {
    log.warn("Kafka publish failed (Best-effort): {}", e.getMessage());
    // 응답에는 영향 없음
}

5.6서비스 간 통신 — RestClient + Resilience4j
⚠️ BFF가 4개 서비스 호출 시 RestClient + Circuit Breaker 필수. CallNotPermittedException → 503 매핑.
// bff-aggregator/.../infrastructure/client/AuthServiceClient.java
@Component
@RequiredArgsConstructor
public class AuthServiceClient {

    private final RestClient restClient;

    @CircuitBreaker(name = "auth-service", fallbackMethod = "getMeFallback")
    @TimeLimiter(name = "auth-service")
    public UserBrief getMe(Long userId) {
        return restClient.get()
            .uri("http://auth-service.qtai.svc.cluster.local:8080/auth/me")
            .header("X-User-Id", userId.toString())
            .retrieve()
            .body(UserBrief.class);
    }

    private UserBrief getMeFallback(Long userId, Throwable t) {
        log.warn("Auth Service circuit open: {}", t.getMessage());
        return null;  // partial 응답
    }
}

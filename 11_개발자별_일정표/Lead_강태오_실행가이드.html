@Component
public class AuthFilter implements GlobalFilter {
    @Override
    public Mono<Void> filter(ServerWebExchange ex, GatewayFilterChain chain) {
        // 1) 외부에서 들어오는 X-User-Id/Role 헤더 strip (spoofing 방지)
        var req = ex.getRequest().mutate()
            .headers(h -> { h.remove("X-User-Id"); h.remove("X-User-Role"); })
            .build();
        // 2) JWT 검증 후 X-User-Id 재주입
        var token = req.getHeaders().getFirst("Authorization");
        if (token != null && jwtVerifier.verify(token)) {
            var claims = jwtVerifier.parse(token);
            req = req.mutate().header("X-User-Id", claims.getSubject()).build();
        }
        return chain.filter(ex.mutate().request(req).build());
    }
}

@UseCase
public class QtTodayAggregateUseCase {
    public QtTodayResponse execute(Long userId) {
        var qt = bibleClient.getTodayQt();   // 필수
        var journalF = supplyAsync(() ->
            tryOrNull(() -> bibleClient.getTodayJournal(userId)));
        var aiF = supplyAsync(() ->
            tryOrNull(() -> aiClient.getRecentSession(userId)));
        var both = allOf(journalF, aiF).join();
        return new QtTodayResponse(qt, journalF.join(), aiF.join());
    }
    static <T> T tryOrNull(Supplier<T> s) { try { return s.get(); } catch (Exception e) { return null; } }
}
